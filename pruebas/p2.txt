fun factorial(n) {
   n>>0?n*factorial(n-1):1
}
fun potencia(b,e) {
   ~e<>0?1:b*potencia(b,e-1)
}
fun numCombinatorio(n,m) {
   factorial(n)/ (factorial(m)*factorial(n-m))
}
fun fib(n) {
   n<=1?1:
     	fib(n-1)+fib(n-2))
}
